## Post-Processing Regex

Always run this after generation:

```python
import re

def clean_for_tts(text):
    # Remove stage directions [text] and (text)
    text = re.sub(r'\[.*?\]', '', text)
    text = re.sub(r'\(.*?\)', '', text)
    
    # Remove Thai meta-sections
    text = re.sub(r'#{1,6}\s*คำแนะนำ.*?(?=---|$)', '', text, flags=re.DOTALL)
    text = re.sub(r'\*\*💡.*?(?=---|$)', '', text, flags=re.DOTALL)
    text = re.sub(r'#{1,6}\s*ข้อแนะนำ.*?(?=---|$)', '', text, flags=re.DOTALL)
    
    # Remove SCRIPT markers
    text = re.sub(r'\[SCRIPT\]\s*', '', text)
    text = re.sub(r'\[/?SCRIPT\]', '', text)
    
    # Remove markdown tables
    text = re.sub(r'\|[^\n]+\|', '', text)
    
    # Clean up multiple newlines
    text = re.sub(r'\n{3,}', '\n\n', text)
    
    return text.strip()
```
